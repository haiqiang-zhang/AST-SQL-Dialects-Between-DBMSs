--Query--
WITH arrayJoin(range(2)) AS delta SELECT     toDate(time) + toIntervalDay(delta) AS dt FROM (     SELECT toDateTime('2020.11.12 19:02:04') AS time ) ORDER BY dt ASC
--Result--
[(datetime.date(2020, 11, 12),), (datetime.date(2020, 11, 13),)]
