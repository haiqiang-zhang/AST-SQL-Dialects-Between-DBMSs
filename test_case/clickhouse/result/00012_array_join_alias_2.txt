--Query--
SELECT x, a, arr FROM (SELECT arrayJoin(['Hello', 'Goodbye']) AS x, [1, 2, 3] AS arr) ARRAY JOIN arr AS a
--Result--
[('Hello', 1, [1, 2, 3]), ('Hello', 2, [1, 2, 3]), ('Hello', 3, [1, 2, 3]), ('Goodbye', 1, [1, 2, 3]), ('Goodbye', 2, [1, 2, 3]), ('Goodbye', 3, [1, 2, 3])]
