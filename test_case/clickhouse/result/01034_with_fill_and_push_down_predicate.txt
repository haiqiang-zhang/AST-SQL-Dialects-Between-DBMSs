--Query--
SELECT * FROM ( SELECT date_time FROM ( SELECT toDateTime('2019-11-14 22:15:00') AS date_time UNION ALL SELECT toDateTime('2019-11-15 01:15:00') AS date_time ) ORDER BY date_time WITH fill step 900 ) WHERE date_time < toDateTime('2019-11-15 00:15:00')

[(datetime.datetime(2019, 11, 14, 22, 15),), (datetime.datetime(2019, 11, 14, 22, 30),), (datetime.datetime(2019, 11, 14, 22, 45),), (datetime.datetime(2019, 11, 14, 23, 0),), (datetime.datetime(2019, 11, 14, 23, 15),), (datetime.datetime(2019, 11, 14, 23, 30),), (datetime.datetime(2019, 11, 14, 23, 45),), (datetime.datetime(2019, 11, 15, 0, 0),)]
