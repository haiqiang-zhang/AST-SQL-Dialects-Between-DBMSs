--Query--
SELECT PLUGIN_NAME, PLUGIN_STATUS, LOAD_OPTION FROM INFORMATION_SCHEMA.PLUGINS   WHERE PLUGIN_NAME IN ('ARCHIVE', 'BLACKHOLE', 'FEDERATED')   ORDER BY PLUGIN_NAME
--Result--
[('ARCHIVE', 'ACTIVE', 'ON'), ('BLACKHOLE', 'ACTIVE', 'ON'), ('FEDERATED', 'DISABLED', 'OFF')]
