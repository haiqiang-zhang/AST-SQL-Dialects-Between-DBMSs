--Query--
select
    *,
    array_agg(col) over(partition by id order by col collate nocase) as lead_col_nocase
from (
	select 
		unnest(array[1, 1, 1, 1]) as id, 
		unnest(array['A', 'a', 'b', 'B']) as col
)
--Result--
[(1, 'A', ['A', 'a']), (1, 'a', ['A', 'a']), (1, 'b', ['A', 'a', 'b', 'B']), (1, 'B', ['A', 'a', 'b', 'B'])]
