SELECT * FROM integers ORDER BY * DESC
--Result--
[(1, None), (1, 3), (0, 2), (0, 1)]
+--------------------+
--Query--
SELECT * FROM integers ORDER BY * DESC NULLS LAST

[(1, 3), (1, None), (0, 2), (0, 1)]
+--------------------+
