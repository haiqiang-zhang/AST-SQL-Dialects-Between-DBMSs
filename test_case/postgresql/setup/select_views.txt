CREATE TABLE customer (
       cid      int primary key,
       name     text not null,
       tel      text,
       passwd	text
);
CREATE TABLE credit_card (
       cid      int references customer(cid),
       cnum     text,
       climit   int
);
CREATE TABLE credit_usage (
       cid      int references customer(cid),
       ymd      date,
       usage    int
);
